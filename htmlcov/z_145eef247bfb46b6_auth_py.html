35 Авторизация пользователя.

37 """ 

38 row = await conn.fetchrow("SELECT id, password_hash FROM users WHERE username = $1", form_data.username) 

39 

40 if not row or not await run_in_threadpool(verify_password, form_data.password, row["password_hash"]): 

41 raise HTTPException( 

42 status_code=status.HTTP_401_UNAUTHORIZED, 

43 detail="Incorrect username or password", 

44 headers={"WWW-Authenticate": "Bearer"}, 

45 ) 

46 

47 access_token = create_access_token(data={"sub": str(row["id"])}) 

48 return {"access_token": access_token, "token_type": "bearer"}
